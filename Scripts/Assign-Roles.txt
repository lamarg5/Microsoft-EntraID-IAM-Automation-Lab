  Connect-MgGraph -Scopes "RoleManagement.ReadWrite.Directory","User.Read.All"

   $Assignments = @(
     @{
         User = "alisonbrown@lamargoodhallgmail.onmicrosoft.com"
         Role = "User Administrator"
     },

     @{
       User = "markkent@lamargoodhallgmail.onmicrosoft.com"
       Role = "User Administrator"
     }
  )

  foreach ($Assignment in $Assignments) {

     try {

        $User = Get-MgUser `
         -Filter "userPrincipalName eq '$($Assignment.User)'"

        $Role = Get-MgDirectoryRole |
        Where-Object {$_.DisplayName -eq $Assignment.Role}

         New-MgDirectoryRoleMemberByRef `
        -DirectoryRoleId $Role.Id `
         -BodyParameter @{
             "@odata.id"="https://graph.microsoft.com/v1.0/directoryObjects/$($User.Id)"
         }

       Write-Host "Assigned $($Assignment.Role) to $($Assignment.User)"
     }

     catch {
         Write-Host $_.Exception.Message
     }
 }
