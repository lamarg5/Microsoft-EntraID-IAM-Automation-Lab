 $UserUPN = "johnsmith@lamargoodhallgmail.onmicrosoft.com"

# Get user
 $User = Get-MgUser -Filter "userPrincipalName eq '$UserUPN'"

 # Disable account
 Update-MgUser `
 -UserId $User.Id `
 -AccountEnabled:$false

 Write-Host "Account disabled for $UserUPN"

 # Remove user from all groups
 $Groups = Get-MgUserMemberOf -UserId $User.Id

 foreach ($Group in $Groups) {

     Remove-MgGroupMemberByRef `
     -GroupId $Group.Id `
     -DirectoryObjectId $User.Id

    Write-Host "Removed from group: $($Group.Id)"
 }

 # Revoke active sign-in sessions
 Revoke-MgUserSignInSession `
 -UserId $User.Id

 Write-Host "Sign-in sessions revoked for $UserUPN"

 Write-Host "Offboarding completed."
