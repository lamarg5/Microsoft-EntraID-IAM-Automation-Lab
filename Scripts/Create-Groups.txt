
Connect-MgGraph -Scopes "Group.ReadWrite.All"

# Groups to create

$Groups = @(
    @{
        DisplayName = "HelpDesk-Team"
        Description = "Help Desk support staff access group"
        MailNickname = "helpdeskteam"
    },

    @{
        DisplayName = "HR-Team"
        Description = "Human Resources access group"
        MailNickname = "hrteam"
    },

    @{
        DisplayName = "IT-Admins"
        Description = "IT administrative access group"
        MailNickname = "itadmins"
    }
)

foreach ($Group in $Groups) {

    try {

        New-MgGroup `
        -DisplayName $Group.DisplayName `
        -Description $Group.Description `
        -MailEnabled:$false `
        -MailNickname $Group.MailNickname `
        -SecurityEnabled:$true

        Write-Host "Created group: $($Group.DisplayName)"

    }

    catch {

        Write-Host "Failed to create: $($Group.DisplayName)"
        Write-Host $_.Exception.Message

    }
}
